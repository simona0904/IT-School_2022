
format_spec ::= [[fill] [align] [sign] [#] [0] [width] [grouping_option] [.precision] [type]
fill ::= <any character>
align ::= ">"  "<"  "="  "^"
sign ::= "+"  "-"  " "
width ::= integer
grouping_option ::= "_"  ","
precision ::= integer
type ::= "b"  "c"  "d"  "e"  "E"  "f"  "F"  "g"  "G"  "n"  "o"  "s"  "x"  "X"  "%"
